\begin{cases} 
LS  	&\|\Delta\|_2

Hugger & \sum\rho_{Huger}(r), 
    \rho_H = \begin{cases}
    r^2/2 &(|r| \le \eta)\\
    \eta|r|-\eta^2/2  (|r| > \eta)
    \end{cases}
\\

Tukey  & \sum\rho_{Tukey}(r),
    \rho_T =  \begin{cases}
        \frac{\left(1- [1-\frac{r^2}{\eta^2}]^3 \right)\eta^2}{6},

    \end{cases}
